Сравнить два числа.

Задание.

Составить программу, в которой вводятся два числа и которая выводит сообщение:

  • о равенстве чисел (если они равны),
  • о том, что первое число меньше втрого (если оно меньше).
  • о том, что первое число больше втрого (если оно больше).

Решение.

#include <iostream>
#include "Windows.h"
using namespace std;
 
// сравнить два числа
void test_005()
{
    int n1, n2;
    cout << "\t\t\tСравнение чисел\n\n";
    cout << "Введите 1 число: ";
    cin >> n1;
    cout << endl;
    cout << "Введите 2 число: ";
    cin >> n2;
    cout << endl;

    if (n1 == n2) {
        cout << "\n\tчисла равны " << n1 << endl;
    } else if (n1 < n2) {
        cout << "\n\tпервое число " << n1 << " меньше второго " << n2 << endl;
    } else {
        cout << "\n\tпервое число " << n1 << " больше второго " << n2 << endl;
    }
}

int main(){
    test_005();
     
    return 0;
}
