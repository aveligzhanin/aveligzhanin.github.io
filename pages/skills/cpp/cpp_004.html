Нарисовать домик.

Составить программу, в которой выводится домик.

#include <iostream>
using namespace std;
 
int main(){
    cout << "t   House:\n";
    cout << "t     /\\\n";
    cout << "t    /  \\\n";
    cout << "t   /    \\\n";
    cout << "t  /------\\\n";
    cout << "t / |    | \\\n";
    cout << "t   |    | \n";
    cout << "t   |    | \n";
    cout << "t   |____| \n";
     
    return 0;
}
